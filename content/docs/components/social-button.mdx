## Usage

```tsx
import { SocialButton } from "@/components/ui/social-button"
```

```tsx
<SocialButton social="google" variant="google">Sign in with Google</SocialButton>
```

## Link

You can use the `asChild` prop to make another component look like a button. Here's an example of a link that looks like a button.

```tsx showLineNumbers
import { Link } from "next/link"

import { SocialButton } from "@/components/ui/social-button"

export function LinkAsButton() {
  return (
    <SocialButton asChild>
      <Link href="/login">Login</Link>
    </SocialButton>
  )
}
```
